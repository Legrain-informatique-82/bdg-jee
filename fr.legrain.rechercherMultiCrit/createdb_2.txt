CREATE TABLE TA_LIGNE_REQUETE (
    ID          DID3,
    ID_RQT          DID3,
	ID_GROUPE		DID3,
	ANDOR_GROUPE	DLGR_2,
	ID_LIGNE		DID3,
	TYPE_LIGNE		DLIB50 NOT NULL,
	ANDOR_LIGNE		DLGR_2,
	COMBO_1			DLIB50 NOT NULL,
	COMBO_2			DLIB50 NOT NULL,
	VALEUR_1		DLIB50,
	VALEUR_2		DLIB50
	
);

ALTER TABLE TA_LIGNE_REQUETE ADD PRIMARY KEY (ID);
ALTER TABLE TA_LIGNE_REQUETE ADD FOREIGN KEY (ID_RQT) REFERENCES TA_REQUETE (ID_RQT);


CREATE GENERATOR NUM_ID_LIGNE_RQT;
SET GENERATOR NUM_ID_LIGNE_RQT TO 0;

SET TERM ^;

CREATE TRIGGER TBID_LIGNE_RQT FOR TA_LIGNE_REQUETE
ACTIVE BEFORE INSERT POSITION 1
as
begin
   If (New.ID  is null) Then
      New.ID  = GEN_ID(NUM_ID_LIGNE_RQT,1);
end
